#9
let userInput = prompt("숫자를 입력하세요.:");
let num = Number(userInput);

let sum = 0;
let i = 1;

while (i <= num) {
    sum += i;
    i++;
}

alert("1부터 " + userInput + "까지의 합은 " + sum + "입니다.");

prompt()는 입력받은 값을 문자열로 저장하므로, 수식적 계산을 위해서는 문자형을 숫자형으로 바꿔줘야 한다.
1부터 입력받은 수까지의 합을 구해야 하므로 반복문을 이용해야 한다.
1부터 순서대로 수를 더하고 반복의 횟수를 결정해주기 위해서는 반복을 위한 변수 i를 설정해주어야 한다.
또한 i가 1부터 입력받은 수까지 1씩 커지는 동안, 해당 단계의 i값을 더하여 저장해줄 변수가 필요하다.
따라서 1부터 입력받은 수까지 모두 더한 값을 나타내기 위한 sum 변수를 설정해야 한다.

#10
(1) a=5, arr=["old", 10, 233, true, "new"]

.push()는 배열의 맨 뒤에 새로운 요소를 추가한 후, 배열에 총 몇 개의 데이터가 있는지 그 결과를 반환하는 메서드이다.
따라서, let a = arr.push("new"); 실행 후 결과는 a=5, arr=["old", 10, 233, true, "new"]가 된다.

(2) b = 3, arr =[10, 233, true]

.shift()는 배열의 맨 앞 요소를 하나 제거하고, 배열에 총 몇 개의 데이터가 있는지 그 결과를 반환하는 메서드이다.
따라서 let b = arr.shift(); 실행 후 결과는 b = 3, arr =[10, 233, true]가 된다.

#13 
let count = 1;
function change() {
    count++;
    const span = document.getElementById("name");
    span.textContent = "likelion_" + count;
}

클릭을 누름에 따라 클릭을 누른 횟수가 likelion_n에서 n의 자리에 반영되어 나타나는 코드다.
따라서, 클릭 버튼을 누른 횟수를 저장하는 변수를 별도로 지정해주어야 하고, 
클릭을 누른 횟수(클릭을 누를 때마다 변하는 값)를 반영하여 html의 body 태그 안의 span 태그에 나타내야 한다.
따라서 클릭 버튼 누른 횟수를 위한 변수 count를 설정하고, .textContent를 이용하여 화면에 나타날 수 있도록 한다.

#14
(1) const btn = document.getElementById("btn");
    btn.addEventListner("click", () => {
        alert("버튼이 클릭됨!")
    })
    
(2) const btn = document.getElementById("btn");
    btn.onclick = () => {
        alert("버튼이 클릭됨!");
    }
이벤트 함수 등록 방식은 .addEventListner()와 .onclick()이 있다. 
onclick()은 속성 값이기에, "click"을 따로 설정하지 않아도 그 자체로 클릭의 실행을 포함한다.
반면, .addEventListner()는 함수이기 때문에 사건의 종류와 함수를 별도로 지정해주어야 한다. 
=> "click"이 있는 첫번째 상자에 .addEventListner를 작성해야 한다.
